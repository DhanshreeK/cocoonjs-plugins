208          * Shows a modal achievements view using a platform dependant view.
209          * @param {function} [callback] The callback function invoked when the modal achievements view is closed by the user. Response params: error.
210          * @see setTemplates
211          */
212         showAchievements : function(callback) {
213             if (!this._achievementsTemplate)
214                 throw "Please, provide a html template for achievements with the setTemplates method";
215             var dialog = new CocoonJS.App.WebDialog();
216             var callbackSent = false;
217             dialog.show(this._achievementsTemplate, function(error) {
218                 dialog.closed = true;
219                 if (!callbackSent && callback)
220                     callback(error);
221             });
222             var me = this;
223             this.requestAchievements(function(achievements, error){
224                 if (dialog.closed)
225                     return;
226                 if (error) {
227                     callbackSent = true;
228                     if (callback)
229                         callback(error);
230                     return;
231                 }
232 
233                 var achs = [];
234                 if (me._cachedAchievements) {
235                     for (var i = 0; i < me._cachedAchievements.length; ++i) {
236                         var ach = me._cachedAchievements[i];
237                         achs.push(ach);
238                         if (achievements && achievements.length) {
239                             for (var j = 0; j< achievements.length; ++j) {
240                                 if (achievements[j].achievementID === ach.achievementID) {
241                                     ach.achieved = true;
242                                     break;
243                                 }
244                             }
245                         }
246                     }
247                 }
248                 var js = "addAchievements(" + JSON.stringify(achs) + ");";
249                 dialog.eval(js);
250             });
251         },

305         /**
306          * @ignore
307          */
308         translateAchievementID: function(id) {
309             if (this._achievementsMap) {
310                 for (var customID in this._achievementsMap) {
311                     if (customID == id) { //important: compare with double equal, because id can be numeric
312                         return this._achievementsMap[customID];
313                     }
314                 }
315             }
316             return id;
317         },
318         /**
319          * @ignore
320          * @param {array} achievements array of achievements
321          */
322         syncAchievementsMap: function(achievements) {
323             if (!this._achievementsMap)
324                 return;
325             for (var i = 0; i< achievements.length; ++i) {
326                 for (var customID in this._achievementsMap) {
327                      if (this._achievementsMap[customID] === achievements[i].achievementID) {
328                          achievements[i].customID = customID;
329                      }
330                 }
331             }
332 
333         }
