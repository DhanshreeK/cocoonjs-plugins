 13     CocoonJS.Social.GameCenterExtension = function() {
 14         this.nativeExtensionName = "IDTK_SRV_GAMECENTER";
 15         this.extensionName = "Social.GameCenter";
 16         this.nativeExtensionObjectAvailable =  CocoonJS.nativeExtensionObjectAvailable && typeof window.ext[this.nativeExtensionName] !== 'undefined';
 17 
 18         var me = this;
 19         if (this.nativeExtensionObjectAvailable) {
 20             this.onGameCenterLoginStateChanged = new CocoonJS.EventHandler(this.nativeExtensionName, this.extensionName, "onGameCenterLoginStateChanged");
 21 
 22             this.onGameCenterLoginStateChanged.addEventListener(function(localPlayer,error){
 23                 me._currentPlayer = localPlayer;
 24             });
 25         }
 26         return this;
 27     };

451     CocoonJS.Social.SocialGamingServiceGameCenter.prototype =  {
452 
453         _cachedAchievements: null,
454 
455         isLoggedIn: function() {
456             return this.gc.isLoggedIn();
457         },
458         login : function(callback) {
459             this.gc.login(function(localPlayer, error){
460                 if (callback)
461                     callback(localPlayer.isAuthenticated, error);
462             });
463         },
464         logout: function(callback) {
465             if (callback)
466                 callback({message:"User has to logout from the Game Center APP"});
467         },
468         getLoggedInUser : function() {
469             return fromGCPLayerToCocoonUser(this.gc._currentPlayer ? this.gc._currentPlayer : this.gc.getLocalPlayer());
470         },
471 
472         requestUser: function(callback, userId) {
473             if (userId) {
474                 this.gc.loadPlayers([userId], function(response, error) {
475                     var user = response && response.length ? fromGCPLayerToCocoonUser(response[0]) : null;
476                     callback(user, error);
477                 });
478             }
479             else {
480                 var me = this;
481                 setTimeout(function(){
482                     callback(me.getLoggedInUser());
483                 })
484             }
485         },
486         requestFriends: function(callback, userId) {
487             this.gc.loadFriends(function(friends, error){
488                 var users = [];
489                 if (friends && friends.length) {
490                     for (var i= 0; i< friends.length; ++i) {
491                         users.push(fromGCPLayerToCocoonUser(friends[i]));
492                     }
493                 }
494                 callback(users, error);
495             });
496         },
497         requestScore: function(callback, params) {
498             var query = {};
499             var options = params || {};
500             if (options.leaderboardID)
501                 query.category = options.leaderboardID;
502             query.playerIDs = [options.userID || this.getLoggedInUser().userID];
503 
504             this.gc.loadScores(function(response, error) {
505                 var gcScore = null;
506                 if (options.userID && response && response.scores && response.scores.length)
507                     gcScore = response.scores[0];
508                 else if (response && response.localPlayerScore)
509                     gcScore = response.localPlayerScore;
510                 var loadedScore = gcScore ? new CocoonJS.Social.Score(gcScore.playerID, gcScore.value, "","", gcScore.category) : null;
511                 callback(loadedScore,error);
512             }, query);
513         },
514 
515         submitScore: function(score, callback, params) {
516             var options = params || {};
517             this.gc.submitScore({value:score, category:options.leaderboardID || ""}, function(error){
518                 if (callback)
519                     callback(error);
520             });
521         },
522 
523         showLeaderboard : function(callback, params) {
524             var options = params || {};
525             this.gc.showLeaderboards(function(error){
526                 if (callback)
527                     callback(error);
528             }, {category:options.leaderboardID || ""});
529         },
530         //internal utility function
531         prepareAchievements: function(reload, callback) {
532 
533             if (!this._cachedAchievements || reload) {
534                 var me = this;
535                 this.gc.loadAchievementDescriptions(function(response, error){
536                     if (error) {
537                         callback([],error);
538                     }
539                     else {
540                        var achievements = [];
541                        if (response && response.length) {
542                            for (var i = 0; i < response.length; i++) {
543                                achievements.push(fromGCAchievementDescriptionToCocoonAchievement(response[i]))
544                            }
545                        }
546                        me.setCachedAchievements(achievements);
547                        callback(achievements, null);
548                     }
549 
550                 });
551             }
552             else {
553                 callback(this._cachedAchievements, null);
554             }
555         },
556 
557         requestAllAchievements : function(callback) {
558             this.prepareAchievements(true, callback);
559         },
560 
561         requestAchievements : function(callback, userID) {
562             var me = this;
563             this.prepareAchievements(false, function(allAchievements, error){
564                 if (error) {
565                     callback([], error);
566                     return;
567                 }
568                 me.gc.loadAchievements(function(response, error){
569                     if (!error) {
570                         var achievements = [];
571                         if (response && response.length) {
572                             for (var i = 0; i < response.length; i++) {
573                                 var ach = me.findAchievement(response[i].identifier);
574                                 if (ach)
575                                     achievements.push(ach);
576                             }
577                         }
578                         callback(achievements, null);
579                     }
580                     else {
581                         callback([], response.error);
582                     }
583 
584                 });
585             });
586         },
587         submitAchievement: function(achievementID, callback) {
588             if (achievementID === null || typeof achievementID === 'undefined')
589                 throw "No achievementID specified";
590             var achID = this.translateAchievementID(achievementID);
591             this.gc.submitAchievements([{identifier:achID, percentComplete:100}], callback);
592         },
593         resetAchievements : function(callback) {
594             this.gc.resetAchievements(callback);
595         },
596         showAchievements : function(callback) {
597             this.gc.showAchievements(function(error){
598                 if (callback)
599                     callback(error);
600             });
601         }
602     }
