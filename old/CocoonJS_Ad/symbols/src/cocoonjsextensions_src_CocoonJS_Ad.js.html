 71 	CocoonJS.Ad.Banner = function(id)
 72 	{
 73 		if (typeof id !== 'number') throw "The given ad ID is not a number.";
 74 
 75 		this.id = id;
 76 		var me = this;
 77 
 78 		/**
 79     	* This {@link CocoonJS.EventHandler} object allows listening to events called when a banner is shown.
 80     	* The callback function does not receive any parameter.
 81     	* @static
 82     	* @event
 83     	*/
 84 		this.onBannerShown = new CocoonJS.EventHandler("IDTK_SRV_AD", "Ad", "onbannershow", function(sourceListener, args)
 85     	{
 86     		if (me.id === args[0]) 
 87     		{
 88     			sourceListener();
 89     		}
 90     	});
 91 
 92     	/**
 93     	* This {@link CocoonJS.EventHandler} object allows listening to events called when a banner is hidden.
 94     	* The callback function does not receive any parameter.
 95     	* @static
 96     	* @event
 97     	*/
 98 		this.onBannerHidden = new CocoonJS.EventHandler("IDTK_SRV_AD", "Ad", "onbannerhide", function(sourceListener, args)
 99     	{
100     		if (me.id === args[0]) 
101     		{
102     			sourceListener();
103     		}
104     	});
105 
106     	/**
107     	* This {@link CocoonJS.EventHandler} object allows listening to events called when a banner is ready (cached).
108     	* The callback function does not receive any parameter.
109     	* @param {number} width The banner width
110     	* @param {number} height The banner height
111     	* @static
112     	* @event
113     	*/
114 		this.onBannerReady = new CocoonJS.EventHandler("IDTK_SRV_AD", "Ad", "onbannerready", function(sourceListener, args)
115     	{
116     		if (me.id === args[0]) 
117     		{
118     			sourceListener(args[1], args[2]);
119     		}
120     	});
121 	};
