 83     /**
 84      * IMPORTANT: This function should only be used by Ludei.
 85      * This function allows a call to the native extension object function reusing the same arguments object.
 86      * Why is interesting to use this function instead of calling the native object's function directly?
 87      * As the CocoonJS object functions expicitly receive parameters, if they are not present and the native call is direcly mapped,
 88      * undefined arguments are passed to the native side. Some native functions do not check the parameters validation
 89      * correctly (just check the number of parameters passed).
 90      * Another solution instead of using this function call is to correctly check if the parameters are valid (not undefined) to make the
 91      * call, but it takes more work than using this approach.
 92      * @static
 93      * @param {string} nativeExtensionObjectName The name of the native extension object name. The object that is a member of the 'ext' object.
 94      * @param {string} nativeFunctionName The name of the function to be called inside the native extension object.
 95      * @param {object} arguments The arguments object of the CocoonJS extension object function. It contains all the arguments passed to the CocoonJS extension object function and these are the ones that will be passed to the native call too.
 96      * @param {boolean} [async] A flag to indicate if the makeCall (false or undefined) or the makeCallAsync function should be used to perform the native call.
 97      * @returns Whatever the native function call returns.
 98      */
 99     CocoonJS.makeNativeExtensionObjectFunctionCall = function(nativeExtensionObjectName, nativeFunctionName, args, async) {
100         if (CocoonJS.isNativeExtensionObjectAvailable()) {
101             var argumentsArray = Array.prototype.slice.call(args);
102             argumentsArray.unshift(nativeFunctionName);
103             var nativeExtensionObject = ext[nativeExtensionObjectName];
104             var makeCallFunction = async ? nativeExtensionObject.makeCallAsync : nativeExtensionObject.makeCall;
105             var ret = makeCallFunction.apply(nativeExtensionObject, argumentsArray);
106             var finalRet = ret;
107             if (typeof ret === "string") {
108                 try {
109                     finalRet = JSON.parse(ret);
110                 } catch (error) {}
111             }
112             return finalRet;
113         }
114     };

116     /**
117      * Returns an object retrieved from a path specified by a dot specified text path starting from a given base object.
118      * It could be useful to find the reference of an object from a defined base object. For example the base object could be window and the
119      * path could be "CocoonJS.App" or "document.body".
120      * @param {Object} baseObject The object to start from to find the object using the given text path.
121      * @param {string} objectPath The path in the form of a text using the dot notation. i.e. "document.body"
122      * For example:
123      * var body = getObjectFromPath(window, "document.body");
124      */
125     CocoonJS.getObjectFromPath = function(baseObject, objectPath) {
126         var parts = objectPath.split('.');
127         var obj = baseObject;
128         for (var i = 0, len = parts.length; i < len; ++i) {
129             obj[parts[i]] = obj[parts[i]] || undefined;
130             obj = obj[parts[i]];
131         }
132         return obj;
133     };

135     /**
136      * Returns the key for a value in a dictionary. It looks for an specific value inside a dictionary and returns the corresponding key.
137      * @param {Object} dictionary The dictionary to look for the value in and get the corresponding key.
138      * @param {Object} value The value to look for inside the dictionary and return it's corresponding key.
139      * @return The key that corresponds to the given value it is has been found or null.
140      */
141     CocoonJS.getKeyForValueInDictionary = function(dictionary, value) {
142         var finalKey = null;
143         for (var key in dictionary) {
144             if (dictionary[key] === value) {
145                 finalKey = key;
146                 break;
147             }
148         }
149         return finalKey;
150     }

152     /**
153      * Finds a string inside a given array of strings by looking for a given substring. It can also
154      * specify if the search must be performed taking care or not of the case sensitivity.
155      * @param {Array} stringArray The array of strings in which to to look for the string.
156      * @param {string} subString The substring to look for inside all the strings of the array.
157      * @param {boolean} caseSensitive A flag to indicate if the search must be performed taking case of the
158      * case sensitiveness (true) or not (false).
159      * @return {string} The string that has been found or null if the substring is not inside any of the string of the array.
160      */
161     CocoonJS.findStringInStringArrayThatIsIndexOf = function(stringArray, subString, caseSensitive) {
162         var foundString = null;
163         subString = caseSensitive ? subString : subString.toUpperCase();
164         for (var i = 0; foundString == null && i < stringArray.length; i++) {
165             foundString = caseSensitive ? stringArray[i] : stringArray[i].toUpperCase();
166             foundString = foundString.indexOf(subString) >= 0 ? stringArray[i] : null;
167         }
168         return foundString;
169     };

171     /**
172      * A class that represents objects to handle events. Event handlers have always the same structure:
173      * Mainly they provide the addEventListener and removeEventListener functions.
174      * Both functions receive a callback function that will be added or removed. All the added callback
175      * functions will be called when the event takes place.
176      * Additionally they also allow the addEventListenerOnce and notifyEventListeners functions.
177      * @constructor
178      * @param {string} nativeExtensionObjectName The name of the native extension object (inside the ext object) to be used.
179      * @param {string} cocoonJSExtensionObjectName The name of the sugarized extension object.
180      * @param {string} nativeEventName The name of the native event inside the ext object.
181      * @param {number} [chainFunction] An optional function used to preprocess the listener callbacks. This function, if provided,
182      * will be called before any of the other listeners.
183      */
184     CocoonJS.EventHandler = function(nativeExtensionObjectName, cocoonJSExtensionObjectName, nativeEventName, chainFunction) {
185         this.listeners = [];
186         this.listenersOnce = [];
187         this.chainFunction = chainFunction;
188 
189         /**
190          * Adds a callback function so it can be called when the event takes place.
191          * @param {function} listener The callback function to be added to the event handler object. See the referenced Listener function documentation for more detail in each event handler object's documentation.
192          */
193         this.addEventListener = function(listener) {
194             if (chainFunction) {
195                 var f = function() {
196                     chainFunction.call(this, arguments.callee.sourceListener, Array.prototype.slice.call(arguments));
197                 };
198                 listener.CocoonJSEventHandlerChainFunction = f;
199                 f.sourceListener = listener;
200                 listener = f;
201             }
202 
203             var cocoonJSExtensionObject = CocoonJS.getObjectFromPath(CocoonJS, cocoonJSExtensionObjectName);
204             if (cocoonJSExtensionObject && cocoonJSExtensionObject.isNativeExtensionObjectAvailable()) {
205                 ext[nativeExtensionObjectName].addEventListener(nativeEventName, listener);
206             }
207             var indexOfListener = this.listeners.indexOf(listener);
208             if (indexOfListener < 0) {
209                 this.listeners.push(listener);
210             }
211         };
212 
213         this.addEventListenerOnce = function(listener) {
214             if (chainFunction) {
215                 var f = function() {
216                     chainFunction(arguments.callee.sourceListener, Array.prototype.slice.call(arguments));
217                 };
218                 f.sourceListener = listener;
219                 listener = f;
220             }
221 
222             var cocoonJSExtensionObject = CocoonJS.getObjectFromPath(CocoonJS, cocoonJSExtensionObjectName);
223             if (cocoonJSExtensionObject.isNativeExtensionObjectAvailable()) {
224                 ext[nativeExtensionObjectName].addEventListenerOnce(nativeEventName, listener);
225             } else {
226                 var indexOfListener = this.listeners.indexOf(listener);
227                 if (indexOfListener < 0) {
228                     this.listenersOnce.push(listener);
229                 }
230             }
231         };
232 
233         /**
234          * Removes a callback function that was added to the event handler so it won't be called when the event takes place.
235          * @param {function} listener The callback function to be removed from the event handler object. See the referenced Listener function documentation for more detail in each event handler object's documentation.
236          */
237         this.removeEventListener = function(listener) {
238 
239             if (chainFunction) {
240                 listener = listener.CocoonJSEventHandlerChainFunction;
241                 delete listener.CocoonJSEventHandlerChainFunction;
242             }
243 
244             var cocoonJSExtensionObject = CocoonJS.getObjectFromPath(CocoonJS, cocoonJSExtensionObjectName);
245             if (cocoonJSExtensionObject.isNativeExtensionObjectAvailable()) {
246                 ext[nativeExtensionObjectName].removeEventListener(nativeEventName, listener);
247             } else {
248                 var indexOfListener = this.listeners.indexOf(listener);
249                 if (indexOfListener >= 0) {
250                     this.listeners.splice(indexOfListener, 1);
251                 }
252             }
253         };
254 
255         this.notifyEventListeners = function() {
256             var cocoonJSExtensionObject = CocoonJS.getObjectFromPath(CocoonJS, cocoonJSExtensionObjectName);
257             var argumentsArray = Array.prototype.slice.call(arguments);
258             if (cocoonJSExtensionObject && cocoonJSExtensionObject.isNativeExtensionObjectAvailable()) {
259                 ext[nativeExtensionObjectName].notifyEventListeners(nativeEventName, argumentsArray);
260             } else {
261                 var listeners = Array.prototype.slice.call(this.listeners);
262                 var listenersOnce = Array.prototype.slice.call(this.listenersOnce);
263                 var _this = this;
264                 // Notify listeners after a while ;) === do not block this thread.
265                 setTimeout(function() {
266                     for (var i = 0; i < listeners.length; i++) {
267                         listeners[i].apply(_this, argumentsArray);
268                     }
269                     for (var i = 0; i < listenersOnce.length; i++) {
270                         listenersOnce[i].apply(_this, argumentsArray);
271                     }
272                 }, 0);
273 
274                 _this.listenersOnce = [];
275             }
276         };
277         return this;
278     };

280     /**
281      * A simple timer class. Update it every loop iteration and get values from accumulated time to elapsed time in seconds or milliseconds.
282      */
283     CocoonJS.Timer = function() {
284         this.reset();
285         return this;
286     };
287 
288     CocoonJS.Timer.prototype = {
289         currentTimeInMillis: 0,
290         lastTimeInMillis: 0,
291         elapsedTimeInMillis: 0,
292         elapsedTimeInSeconds: 0,
293         accumTimeInMillis: 0,
294 
295         /**
296 		Resets the timer to 0.
297 		*/
298         reset: function() {
299             this.currentTimeInMillis = this.lastTimeInMillis = new Date().getTime();
300             this.accumTimeInMillis = this.elapsedTimeInMillis = this.elapsedTimeInSeconds = 0;
301         },
302 
303         /**
304 		Updates the timer calculating the elapsed time between update calls.
305 		*/
306         update: function() {
307             this.currentTimeInMillis = new Date().getTime();
308             this.elapsedTimeInMillis = this.currentTimeInMillis - this.lastTimeInMillis;
309             this.elapsedTimeInSeconds = this.elapsedTimeInMillis / 1000.0;
310             this.lastTimeInMillis = this.currentTimeInMillis;
311             this.accumTimeInMillis += this.elapsedTimeInMillis;
312         }
313     };
