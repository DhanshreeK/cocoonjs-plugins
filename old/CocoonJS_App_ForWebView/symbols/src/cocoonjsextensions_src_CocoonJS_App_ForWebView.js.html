 67

 88     /**
 89     * This function allows to forward console messages from the WebView to the CocoonJS
 90     * debug console. What it does is to change the console object for a new one
 91     * with all it's methods (log, error, info, debug and warn) forwarding their
 92     * messages to the CocoonJS environment.
 93     * The original console object is stored in the CocoonJS.App.originalConsole property.
 94     * @function
 95     */
 96     CocoonJS.App.proxifyConsole = function () {
 97         if (!CocoonJS.isNativeExtensionObjectAvailable()) return;
 98 
 99         if (typeof CocoonJS.App.originalConsole === 'undefined') {
100             CocoonJS.App.originalConsole = window.console;
101         }
102         var functions = ["log", "error", "info", "debug", "warn"];
103 
104         var newConsole = {};
105         for (var i = 0; i < functions.length; i++) {
106             newConsole[functions[i]] = function (functionName) {
107                 return function (message) {
108                     var jsCode = "console." + functionName + "(" + JSON.stringify(message) + ");";
109                     CocoonJS.App.originalConsole.log(jsCode);
110                     ext.IDTK_APP.makeCallAsync("forward", jsCode);
111                 };
112             }(functions[i]);
113         }
114         if (!newConsole.assert) {
115             newConsole.assert = function assert() {
116                 if (arguments.length > 0 && !arguments[0]) {
117                     var str = 'Assertion failed: ' + (arguments.length > 1 ? arguments[1] : '');
118                     newConsole.error(str);
119                 }
120             }
121         }
122         window.console = newConsole;
123     };

133             CocoonJS.App.originalConsole = undefined;

150             function forwardEventToCocoonJS(eventName, event) {
151                 var eventData = {};
152                 var att, i;
153                 for (var att in event) {
154                     i = EVENT_ATTRIBUTES.indexOf(att);
155                     if (i >= 0) {
156                         eventData[att] = event[att];
157                     }
158                 }
159                 var jsCode = "CocoonJS && CocoonJS.App && CocoonJS.App.forwardedEventFromTheWebView && CocoonJS.App.forwardedEventFromTheWebView(" + JSON.stringify(eventName) + ", '" + JSON.stringify(eventData) + "');";
160                 CocoonJS.App.forward(jsCode);
161             }
162             for (i = 0; i < EVENTS.length; i++) {
163                 window.addEventListener(EVENTS[i], (function (eventName) {
164                     return function (event) {
165                         if (CocoonJS.App.forwardEventsToCocoonJSEnabled) {
