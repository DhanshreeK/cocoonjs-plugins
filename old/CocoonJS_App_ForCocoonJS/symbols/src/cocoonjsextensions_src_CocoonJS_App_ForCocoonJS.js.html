 46     function checkEmulatedWebViewReady() {
 47         var emulatedWB = CocoonJS.App.EmulatedWebView;
 48         if (emulatedWB) {
 49             return; //ready
 50         }
 51 
 52         emulatedWB = document.createElement('div');
 53         emulatedWB.setAttribute('id', 'CocoonJS_App_ForCocoonJS_WebViewDiv');
 54         emulatedWB.style.width = 0;
 55         emulatedWB.style.height = 0;
 56         emulatedWB.style.position = "absolute";
 57         emulatedWB.style.left = 0;
 58         emulatedWB.style.top = 0;
 59         emulatedWB.style.backgroundColor = 'transparent';
 60         emulatedWB.style.border = "0px solid #000";
 61 
 62         var frame = document.createElement("IFRAME");
 63         frame.setAttribute('id', 'CocoonJS_App_ForCocoonJS_WebViewIFrame');
 64         frame.setAttribute('name', 'CocoonJS_App_ForCocoonJS_WebViewIFrame');
 65         frame.style.width = 0;
 66         frame.style.height = 0;
 67         frame.frameBorder = 0;
 68         frame.allowtransparency = true;
 69 
 70         emulatedWB.appendChild(frame);
 71         CocoonJS.App.EmulatedWebView = emulatedWB;
 72         CocoonJS.App.EmulatedWebViewIFrame = frame;
 73 
 74         if (!document.body) {
 75             document.body = document.createElement("body");
 76         }
 77         document.body.appendChild(CocoonJS.App.EmulatedWebView);
 78     }

102     /**
103     * Loads a resource in the WebView environment from the CocoonJS environment.
104     * @function
105     * @param {string} path The path to the resource. It can be a remote URL or a path to a local file.
106     * @param {CocoonJS.App.StorageType} [storageType] An optional parameter to specify at which storage in the device the file path is stored. By default, APP_STORAGE is used.
107     * @see CocoonJS.App.load
108     * @see CocoonJS.App.onLoadInTheWebViewSucceed
109     * @see CocoonJS.App.onLoadInTheWebViewFailed
110     */
111     CocoonJS.App.loadInTheWebView = function (path, storageType) {
112         if (navigator.isCocoonJS && CocoonJS.App.isNativeExtensionObjectAvailable()) {
113             CocoonJS.makeNativeExtensionObjectFunctionCall("IDTK_APP", "loadInTheWebView", arguments)
114         }
115         else {
116             var xhr = new XMLHttpRequest();
117 
118             xhr.onreadystatechange = function (event) {
119                 if (xhr.readyState === 4) {
120                     if ((xhr.status >= 200 && xhr.status <= 299) || xhr.status === 0) {
121 
122                         checkEmulatedWebViewReady();
123                         var callback = function (event) {
124                             CocoonJS.App.onLoadInTheWebViewSucceed.notifyEventListeners(path);
125                             CocoonJS.App.EmulatedWebViewIFrame.removeEventListener("load", callback);
126                         };
127 
128                         CocoonJS.App.EmulatedWebViewIFrame.addEventListener(
129                             "load",
130                             callback
131                         );
132                         CocoonJS.App.EmulatedWebViewIFrame.contentWindow.location.href = path;
133                     }
134                     else {
135                         this.onreadystatechange = null;
136                         CocoonJS.App.onLoadInTheWebViewFailed.notifyEventListeners(path);
137                     }
138                 }
139             };
140             xhr.open("GET", path, true);
141             xhr.send();
142         }
143     };

289             textColor = textColor ? textColor : "white";
290             var jsCode = "" +
291                 "(function()" +

294                 "var loadedScriptCounter = 0;" +
295                 "var loadedScriptFunction = function() " +
296                 "{ " +
297                 "loadedScriptCounter++;" +
298                 "if (loadedScriptCounter >= COCOONJS_WEBVIEW_EXTENSION_SCRIPT_FILES.length)" +
299                 "{" +
300                 "CocoonJS.App.enableFPS(" + JSON.stringify(fpsLayout) + ", " + JSON.stringify(textColor) + ");" +
301                 "}" +
302                 "};" +
303                 "for (var i = 0; i < COCOONJS_WEBVIEW_EXTENSION_SCRIPT_FILES.length; i++)" +
304                 "{" +
305                 "var s = document.createElement('script');" +
306                 "s.onload = loadedScriptFunction;" +
307                 "s.src = COCOONJS_WEBVIEW_EXTENSION_SCRIPT_FILES[i];" +
308                 "document.body.appendChild(s);" +
309                 "}" +
310                 "})();";
311             setTimeout(function () {
312                 CocoonJS.App.forward(jsCode);
313             }, 3000);
314             CocoonJS.App.fpsInTheWebViewEnabled = true;

342     CocoonJS.App.forwardedEventFromTheWebView = function (eventName, eventDataString) {
343         var eventData = JSON.parse(eventDataString);
344         eventData.target = window;
345         var event = new Event(eventName);
346         for (var att in eventData) {
347             event[att] = eventData[att];
348         }
349         event.target = window;
350         window.dispatchEvent(event);
351         var canvases = document.getElementsByTagName("canvas");
352         for (var i = 0; i < canvases.length; i++) {
353             event.target = canvases[i];
354             canvases[i].dispatchEvent(event);
355         }
356     }
