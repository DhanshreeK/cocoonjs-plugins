329                 this[eventKey].push(callback);
330             }

332             {
333                 FB.Event.subscribe(name,callback);

337          * Removes handlers on events so that it no longer invokes your callback when the event fires.
338          * @param name Name of the event.
339          * @param callback The handler function.
340          */
341         unsubscribe: function(name, callback) {
342             if (this.nativeExtensionObjectAvailable)
343             {
344                 var eventKey = name + 'listeners';

347                     var index = array.indexOf(callback);

350                     }
351                 }

400     function fromCocoonFBSessionToFBAPISession(response, error) {
401 
402         var authResponse = null;
403         if (response.state === 0) {
404             authResponse = {
405                 accessToken: response.accessToken,
406                 expirationDate: response.expirationDate,
407                 userID: response.user ? response.userID : null,
408                 permissions: response.permissions,
409                 user: response.user
410             }
411         }
412 
413         return {
414             status: fromCocoonFBStatusToFBAPIState(response.state),
415             authResponse: authResponse,
416             error: error
417         }
