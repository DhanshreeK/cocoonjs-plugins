 33     /**
 34      * The data structure that represents the information of a local notification.
 35      * @namespace
 36      * @constructor
 37      * @deprecated
 38      * @param {string} message The notification message.
 39      * @param {boolean} soundEnabled A flag that indicates if the sound should be enabled for the notification.
 40      * @param {number} badgeNumber The number that will appear in the badge of the application icon in the home screen.
 41      * @param {object} userData The JSON data to attached to the notification.
 42      * @param {string} contentBody The body content to be showed in the expanded notification information.
 43      * @param {string} contentTitle The title to be showed in the expanded notification information.
 44      * @param {number} date Time in millisecs from 1970 when the notification will be fired.
 45      */
 46     CocoonJS.Notification.LocalNotification = function(message, soundEnabled, badgeNumber, userData, contentBody, contentTitle, scheduleTime) {
 47         //The id is initialized with -1 for compatibility issues.
 48         //The new way to use is createLocalNotification. These can be cancelled and the ID is sent from native code to the 
 49         this.id = -1;
 50 
 51         /**
 52          * The notification message.
 53          * @field
 54          * @type string
 55          */
 56         this.message = message;
 57 
 58         /**
 59          * A flag that indicates if the sound should be enabled for the notification.
 60          * @field
 61          * @type boolean
 62          */
 63         this.soundEnabled = soundEnabled;
 64 
 65         /**
 66          * (iOS only) The number that will appear in the badge of the application icon in the home screen.
 67          * @field
 68          * @type number
 69          */
 70         this.badgeNumber = badgeNumber;
 71 
 72         /**
 73          * The JSON data to attached to the notification.
 74          * @field
 75          * @type object
 76          */
 77         this.userData = userData;
 78 
 79         /**
 80          * (Android only) The body content to be showed in the expanded notification information.
 81          * @field
 82          * @type string
 83          */
 84         if (contentBody !== undefined)
 85             this.contentBody = contentBody;
 86         else
 87             this.contentBody = "";
 88 
 89         /**
 90          * (Android only) The title to be showed in the expanded notification information.
 91          * @field
 92          * @type string
 93          */
 94         if (contentTitle !== undefined)
 95             this.contentTitle = contentTitle;
 96         else
 97             this.contentTitle = "";
 98 
 99         /**
100          * Time in millisecs from 1970 when the local notification will be fired
101          * @field
102          * @type number
103          */
104         var currentTime = new Date().getTime();
105         if (scheduleTime !== undefined) {
106             this.scheduleTime = scheduleTime;
107         } else {
108             this.scheduleTime = currentTime;
109         }
110     };

112     /**
113      * The data structure that represents the information of a push notification.
114      * @namespace
115      * @constructor
116      * @param {string} message The notification message.
117      * @param {boolean} soundEnabled A flag that indicates if the sound should be enabled for the notification.
118      * @param {number} badgeNumber The number that will appear in the badge of the application icon in the home screen.
119      * @param {object} userData The JSON data to attached to the notification.
120      * @param {array} channels An array containing the channels names this notification will be delivered to.
121      * @param {number} expirationTime A time in seconds from 1970 when the notification is no longer valid and will not be delivered in case it has not already been delivered.
122      * @param {number} expirationTimeInterval An incremental ammount of time in from now when the notification is no longer valid and will not be delivered in case it has not already been delivered.
123      */
124     CocoonJS.Notification.PushNotification = function(message, soundEnabled, badgeNumber, userData, channels, expirationTime, expirationTimeInterval) {
125         /**
126          * The notification message.
127          * @field
128          * @type string
129          */
130         this.message = message;
131 
132         /**
133          * A flag that indicates if the sound should be enabled for the notification.
134          * @field
135          * @type boolean
136          */
137         this.soundEnabled = soundEnabled;
138 
139         /**
140          * The number that will appear in the badge of the application icon in the home screen.
141          * @field
142          * @type number
143          */
144         this.badgeNumber = badgeNumber;
145 
146         /**
147          * The JSON data to attached to the notification.
148          * @field
149          * @type object
150          */
151         this.userData = userData
152 
153         /**
154          * An array containing the channels names this notification will be delivered to.
155          * @field
156          * @type array
157          */
158         this.channels = channels;
159 
160         /**
161          * A time in seconds from 1970 when the notification is no longer valid and will not be delivered in case it has not already been delivered.
162          * @field
163          * @type number
164          */
165         if (expirationTime !== undefined)
166             this.expirationTime = expirationTime;
167         else
168             this.expirationTime = 0;
169 
170         /**
171          * An incremental ammount of time in from now when the notification is no longer valid and will not be delivered in case it has not already been delivered.
172          * @field
173          * @type number
174          */
175         if (expirationTimeInterval !== undefined)
176             this.expirationTimeInterval = expirationTimeInterval;
177         else
178             this.expirationTimeInterval = 0;
179     };
