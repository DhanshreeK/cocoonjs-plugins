 14 	/**
 15 	* The data structure that represents the information of a local notification.
 16 	* @namespace
 17 	* @constructor
 18 	* @param {string} message The notification message.
 19 	* @param {boolean} soundEnabled A flag that indicates if the sound should be enabled for the notification.
 20 	* @param {number} badgeNumber The number that will appear in the badge of the application icon in the home screen.
 21 	* @param {object} userData The JSON data to attached to the notification.
 22 	* @param {string} contentBody The body content to be showed in the expanded notification information.
 23 	* @param {string} contentTitle The title to be showed in the expanded notification information.
 24 	* @param {number} date Time in millisecs from 1970 when the notification will be fired.
 25 	*/
 26 	CocoonJS.Notification.LocalNotification = function(message, soundEnabled, badgeNumber, userData, contentBody, contentTitle, scheduleTime)
 27 	{
 28 		/**
 29 		* The notification message.
 30 		* @field
 31 		* @type string
 32 		*/
 33 		this.message = message;
 34 
 35 		/**
 36 		* A flag that indicates if the sound should be enabled for the notification.
 37 		* @field
 38 		* @type boolean
 39 		*/
 40 		this.soundEnabled = soundEnabled;
 41 
 42 		/**
 43 		* (iOS only) The number that will appear in the badge of the application icon in the home screen.
 44 		* @field
 45 		* @type number
 46 		*/
 47 		this.badgeNumber = badgeNumber;
 48 
 49 		/**
 50 		* The JSON data to attached to the notification.
 51 		* @field
 52 		* @type object
 53 		*/
 54 		this.userData = userData;
 55 
 56 		/**
 57 		* (Android only) The body content to be showed in the expanded notification information.
 58 		* @field
 59 		* @type string
 60 		*/
 61 		if (contentBody !== undefined)
 62 			this.contentBody = contentBody;
 63 		else
 64 			this.contentBody = "";
 65 
 66 		/**
 67 		* (Android only) The title to be showed in the expanded notification information.
 68 		* @field
 69 		* @type string
 70 		*/
 71 		if (contentTitle !== undefined)
 72 			this.contentTitle = contentTitle;
 73 		else
 74 			this.contentTitle = "";
 75 
 76 		/**
 77 		* Time in millisecs from 1970 when the local notification will be fired
 78 		* @field
 79 		* @type number
 80 		*/
 81 		var currentTime = new Date().getTime();
 82 		if (scheduleTime !== undefined){
 83 			this.scheduleTime = scheduleTime;
 84 		}else{
 85 			this.scheduleTime = currentTime;
 86 		}
 87 	};

 89 	/**
 90 	* The data structure that represents the information of a push notification.
 91 	* @namespace
 92 	* @constructor
 93 	* @param {string} message The notification message.
 94 	* @param {boolean} soundEnabled A flag that indicates if the sound should be enabled for the notification.
 95 	* @param {number} badgeNumber The number that will appear in the badge of the application icon in the home screen.
 96 	* @param {object} userData The JSON data to attached to the notification.
 97 	* @param {array} channels An array containing the channels names this notification will be delivered to.
 98 	* @param {number} expirationTime A time in seconds from 1970 when the notification is no longer valid and will not be delivered in case it has not already been delivered.
 99 	* @param {number} expirationTimeInterval An incremental ammount of time in from now when the notification is no longer valid and will not be delivered in case it has not already been delivered.
100 	*/
101 	CocoonJS.Notification.PushNotification = function(message, soundEnabled, badgeNumber, userData, channels, expirationTime, expirationTimeInterval)
102 	{
103 		/**
104 		* The notification message.
105 		* @field
106 		* @type string
107 		*/
108 		this.message = message;
109 
110 		/**
111 		* A flag that indicates if the sound should be enabled for the notification.
112 		* @field
113 		* @type boolean
114 		*/
115 		this.soundEnabled = soundEnabled;
116 
117 		/**
118 		* The number that will appear in the badge of the application icon in the home screen.
119 		* @field
120 		* @type number
121 		*/
122 		this.badgeNumber = badgeNumber;
123 
124 		/**
125 		* The JSON data to attached to the notification.
126 		* @field
127 		* @type object
128 		*/
129 		this.userData = userData
130 
131 		/**
132 		* An array containing the channels names this notification will be delivered to.
133 		* @field
134 		* @type array
135 		*/
136 		this.channels = channels;
137 
138 		/**
139 		* A time in seconds from 1970 when the notification is no longer valid and will not be delivered in case it has not already been delivered.
140 		* @field
141 		* @type number
142 		*/
143 		if (expirationTime !== undefined)
144 			this.expirationTime = expirationTime;
145 		else
146 			this.expirationTime = 0;
147 
148 		/**
149 		* An incremental ammount of time in from now when the notification is no longer valid and will not be delivered in case it has not already been delivered.
150 		* @field
151 		* @type number
152 		*/
153 		if (expirationTimeInterval !== undefined)
154 			this.expirationTimeInterval = expirationTimeInterval;
155 		else
156 			this.expirationTimeInterval = 0;
157 	};
