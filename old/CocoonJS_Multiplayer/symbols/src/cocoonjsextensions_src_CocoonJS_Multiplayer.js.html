  9 	* This type represents the access to a native Multiplayer extension API.
 10     * This service allows to programmatically create matches to invite other players and to receive match invitations sent by other players.
 11     * Your game must authenticate the related social service before you can use this service.
 12 	* @class
 13 	* @constructor
 14 	* @param {string} nativeExtensionName The name of the native ext object extension property name.
 15 	* @param {string} extensionName The name of the CocoonJS object extension property name.
 16 	*/
 17 	CocoonJS.Multiplayer.MultiplayerService = function(nativeExtensionName, extensionName)
 18 	{
 19 		if (typeof nativeExtensionName !== 'string') throw "The given native extension name '" + nativeExtensionName + "' is not a string.";
 20 		if (typeof extensionName !== 'string') throw "The given extension name '" + nativeExtensionName + "' is not a string.";
 21 
 22 		this.nativeExtensionName = nativeExtensionName;
 23 		this.extensionName = extensionName;
 24 	    this.nativeExtensionObjectAvailable = CocoonJS.nativeExtensionObjectAvailable && typeof window.ext[nativeExtensionName] !== 'undefined';
 25 
 26         var me = this;
 27 
 28         /**
 29          * This {@link CocoonJS.EventHandler} object allows listening to events called when an invitation has been received from another player and the user has accepted it
 30          * The invitation will start loading.
 31          * While the invitation is loading a system view might appear on the screen.
 32          * You should pause/stop your current game and prepare it to process a new match
 33          * @memberOf CocoonJS.Multiplayer.MultiplayerService
 34          */
 35         this.onInvitationReceived = new CocoonJS.EventHandler(this.nativeExtensionName, this.extensionName, "onInvitationReceived");
 36         /**
 37          * This {@link CocoonJS.EventHandler} object allows listening to events called when an accepted invitation has been loaded
 38          * The invitation is ready to be processed (start the match or handle the error)
 39          * @memberOf CocoonJS.Multiplayer.MultiplayerService
 40          * Received params: {CocoonJS.Multiplayer.Match} and error
 41          */
 42         this.onInvitationLoaded = new CocoonJS.EventHandler(this.nativeExtensionName, this.extensionName, "onInvitationLoaded", function(sourceListener, args){
 43             var matchID = args[0];
 44             var error = args[1];
 45             if (matchID && !error) {
 46                 me.currentMatch =  new CocoonJS.Multiplayer.Match(me.nativeExtensionName, me.extensionName, matchID);
 47                 sourceListener(me.currentMatch, null)
 48             }
 49             else {
 50                 sourceListener(null, error);
 51             }
 52         });
 53 
 54 
 55 
 56         return this;
 57 	};

133 	/**
134 	* This type provides a transmission network between a group of users.
135 	* The match might be returned before connections have been established between players. At this stage, all the players are in the process of connecting to each other.
136 	* Always check the getExpectedPlayerCount value before starting a match. When its value reaches zero, all expected players are connected, and your game can begin the match.
137 	* Do not forget to call the start method of the match when your game is ready to process received messages via onMatchDataReceived listener.
138 	* @class
139 	* @constructor
140 	* @param {string} nativeExtensionName The name of the native ext object extension property name.
141 	* @param {string} extensionName The name of the CocoonJS object extension property name.
142 	* @param {number} matchID The match ID user for native service bridge.
143 	*/
144 	CocoonJS.Multiplayer.Match = function(nativeExtensionName, extensionName, matchID)
145 	{
146 		if (typeof nativeExtensionName !== 'string') throw "The given native extension name '" + nativeExtensionName + "' is not a string.";
147 		if (typeof extensionName !== 'string') throw "The given extension name '" + nativeExtensionName + "' is not a string.";
148 
149 		this.nativeExtensionName = nativeExtensionName;
150 		this.extensionName = extensionName;
151 	    this.nativeExtensionObjectAvailable = CocoonJS.nativeExtensionObjectAvailable && typeof window.ext[nativeExtensionName] !== 'undefined';
152 	    this.matchID = matchID;
153 	    var me = this;
154 
155 
156 	    /**
157 	    * This {@link CocoonJS.EventHandler} object allows listening to events called when a match receives data from the network
158 	    * The callback function's documentation is represented by {@link CocoonJS.Social.OnMatchDataReceivedListener}
159 	    * @memberOf CocoonJS.Multiplayer.Match
160 	    * @param {CocoonJS.Multiplayer.Match} match The source match.
161 	    * @param {string} data The received data
162 	    * @param {string} playerID The playerID where the data is received from.
163 	    */
164 	    this.onMatchDataReceived = new CocoonJS.EventHandler(this.nativeExtensionName, this.extensionName, "onMatchDataReceived", function(sourceListener, args) {
165 	    		if (me.matchID === args[0]) {
166 	    			sourceListener(me, args[1], args[2]);
167 	    		}
168 	    	});
169 
170 	    /**
171 	    * This {@link CocoonJS.EventHandler} object allows listening to events called when a player connection state changes.
172 	    * The callback function's documentation is represented by {@link CocoonJS.Social.OnMatchStateChangedListener}
173 	    * @memberOf CocoonJS.Multiplayer.Match
174 	    * @param {CocoonJS.Multiplayer.Match} match The source match.
175 	    * @param {string} playerID The player whose state has changed
176 	    * @param {CocoonJS.Multiplayer.ConnectionState} The new connection state of the player
177 	    */
178 	    this.onMatchStateChanged = new CocoonJS.EventHandler(this.nativeExtensionName, this.extensionName, "onMatchStateChanged", function(sourceListener, args) {
179 	    		if (me.matchID === args[0]) {
180 	    			sourceListener(me, args[1], args[2]);
181 	    		}
182 	    	});
183 
184 	    /**
185 	    * This {@link CocoonJS.EventHandler} object allows listening to events called when a netowrk connection with a player fails
186 	    * The callback function's documentation is represented by {@link CocoonJS.Social.OnMatchConnectionWithPlayerFailedListener}
187 	    * @memberOf CocoonJS.Multiplayer.Match
188 	    * @param {CocoonJS.Multiplayer.Match} match The source match.
189 	    * @param {string} playerID The player whose state has changed
190 	    * @param {string} error The error message
191 	    */
192 	    this.onMatchConnectionWithPlayerFailed = new CocoonJS.EventHandler(this.nativeExtensionName, this.extensionName, "onMatchConnectionWithPlayerFailed", function(sourceListener, args) {
193 	    		if (me.matchID === args[0]) {
194 	    			sourceListener(me, args[1], args[2]);
195 	    		}
196 	    	});
197 
198 	    /**
199 	    * This {@link CocoonJS.EventHandler} object allows listening to events called when the match fails
200 	    * The callback function's documentation is represented by {@link CocoonJS.Social.OnMatchFailedListener}
201 	    * @memberOf CocoonJS.Multiplayer.Match
202 	    * @param {CocoonJS.Multiplayer.Match} match The source match.
203 	    * @param {string} error The error message
204 	    */
205 	    this.onMatchFailed = new CocoonJS.EventHandler(this.nativeExtensionName, this.extensionName, "onMatchFailed", function(sourceListener, args) {
206 	    		if (me.matchID === args[0]) {
207 	    			sourceListener(me, args[1]);
208 	    		}
209 	    	});
210 	};
